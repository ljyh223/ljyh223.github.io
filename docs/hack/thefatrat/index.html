[common]

[attack_machine_reverse_proxy]
type = tcp
local_ip = 127.0.0.1
local_port = 4444  # 攻击机上监听的端口（用来接收来自目标的连接）
remote_port = 6000  # VPS上开放的远程端口，用来转发流量
