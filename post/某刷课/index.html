def RSA_encrypt(clear):
    if clear=='':
        return '' 
    rsa_key = RSA.import_key(base64.b64decode(KEY))
    cipher = PKCS1_v1_5.new(rsa_key)
    return base64.b64encode(cipher.encrypt(str(clear).encode('utf-8'))).decode()
